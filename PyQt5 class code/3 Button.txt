from PyQt5.QtWidgets import QMainWindow,QApplication,QWidget,QPushButton
from PyQt5.QtGui import QIcon
import sys

class Window(QMainWindow):
    def __init__(self):
        super().__init__()

        self.ICON = "icon.jpg"
        self.TITLE="RUET"
        self.X_start=200
        self.Y_start =200
        self.WIDTH=400
        self.HEIGHT=200

        self.INIT_WINDOW()
        self.setCentralWidget(ComponentManagement())
        self.show()

    def INIT_WINDOW(self):
        self.setWindowTitle(self.TITLE)
        self.setWindowIcon(QIcon(self.ICON))
        self.setGeometry(self.X_start,self.Y_start,self.WIDTH,self.HEIGHT)
        self.resize(800, 200)
        self.setFixedSize(self.size())


class ComponentManagement(QWidget):
    def __init__(self):
        super().__init__()

        self.UIcomponent()

    def UIcomponent(self):
        BUTTON_1 = QPushButton("Button 1",self)
        BUTTON_2 = QPushButton("Button 2", self)
        BUTTON_1.move(100,100)

if __name__=="__main__":
    App=QApplication(sys.argv)
    WINDOW=Window()
    sys.exit(App.exec())
